Agregar Controladores un ReportViewer y un Script Manager.


Para llamarlo se usa el nombre de la carpeta que es e l nombre del proyecto donde se creó el reporte:

    public partial class Vista01 : System.Web.UI.Page

    {

        protected void Page_Load(object sender, EventArgs e)

        {

            ShowRep("MvcReportingServices01", "Reporte01CSharp");

        }

 

        protected void ShowRep(string nombrepath, string reporteremoto )

        {

            if (!IsPostBack)

            {

                ReportViewer1.ProcessingMode = ProcessingMode.Remote;

                ReportViewer1.ServerReport.ReportServerUrl = new Uri("http://pcnavigator/ReportServer");

                ReportViewer1.ServerReport.ReportPath = "/" + nombrepath + "/" + reporteremoto;

                ReportViewer1.ServerReport.Refresh();

            }

        }

    }

        function ReportValidationCheck() {

            url = "../Reports/Vista01.aspx";

 

            var myframe = document.getElementById("ifrmReport");

            if (myframe !== null) {

                if (myframe.src) {

                    myframe.src = url;

                }

                else if (myframe.contentWindow !== null && myframe.contentWindow.location !== null) {

                    myframe.contentWindow.location = url;

                }

                else { myframe.setAttribute( 'src' , url ); }

            }

            return false;

        }

    </script>   

   

    }
